/****************************** ** ** Class Emitter ** ******************************/ Dans le fichier de votre class JS : import Emitter from "dauphine/src/scripts/_emitter"; export default class Modal extends Emitter { Bien penser à extends la Class Emitter constructor(el){ super(); this.modal = el; }; init(){ Le premier argument est le nom de l'événement et le second correspond au callback (objet, function...) this.emit('init', { modal: this.modal }); }; }; Dans le fichier d'initialisation de la class : const element = document.querySelector('.modal'); const modalEl = new Modal(element); modalEl.on('init', (callback) => { console.log(callback.modal); callback correspond à notre object qui contient actuellement modal });
